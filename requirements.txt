# Runtime dependencies for the MG400 platform.
#
# Keep this file pure ASCII. pip on Windows reads it with the system locale
# (cp950 for Traditional Chinese installs) and chokes on non-ASCII bytes
# (UnicodeDecodeError). UTF-8 em-dashes etc. are out; use plain hyphens.
#
# Python version: requires 3.10 <= python < 3.13. Reason: DmvSDK (Delta
# industrial camera SDK; needed for M0 calibration and M2 inspection viz)
# ships C extensions for cp38..cp312 only -- newer Python has no matching
# .pyd and import fails with a misleading "circular import" message
# (PROGRESS.md finding 26). Mac/Linux dev without DmvSDK can use newer
# Python freely, but Windows machines that talk to the camera must stay
# in this range. 3.11.9 aligns with phase5-panel for easy package sharing.
#
# numpy parses the 1440-byte binary feedback frame (robot_core/transport/feedback.py).
numpy>=1.24

# viz/ inspection UI backend (ws server). FastAPI for ws + future M3 REST hooks;
# uvicorn[standard] pulls websockets + httptools for production-grade ws handling.
fastapi>=0.110
uvicorn[standard]>=0.27
# Pulled in by fastapi.testclient (TestClient is a Starlette helper).
httpx>=0.27

# Pillow saves camera frames as PNG (robot_core/scripts/camera_smoke.py and
# the M0b/M0c calibration scripts). Already installed via phase5-panel for
# Win operators, but listed explicitly so a fresh venv works without extra
# manual installs.
Pillow>=10.0

# opencv-contrib-python (NOT plain opencv-python) for ChArUco calibration
# board + cv2.aruco. The plain package lacks the contrib aruco submodule;
# co-installing both causes import conflicts so use contrib only.
opencv-contrib-python>=4.10
